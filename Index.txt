  <script>
    function initAutocomplete() {
      const input = document.getElementById('destination');
      new google.maps.places.Autocomplete(input);
    }

    function calculateDeparture() {
      const destination = document.getElementById('destination').value;
      const arrivalTime = document.getElementById('arrival').value;
      const arrivalDay = document.getElementById('arrivalDay').value;
      const resultDiv = document.getElementById('result');

      if (!destination || !arrivalTime) {
        resultDiv.textContent = 'Please fill out all fields.';
        return;
      }

      resultDiv.textContent = 'Calculating...';

      const [hours, minutes] = arrivalTime.split(':').map(Number);
      const now = new Date();
      const arrivalDateTime = new Date();

      if (arrivalDay === 'tomorrow') {
        arrivalDateTime.setDate(now.getDate() + 1);
      }

      arrivalDateTime.setHours(hours);
      arrivalDateTime.setMinutes(minutes);
      arrivalDateTime.setSeconds(0);

      const service = new google.maps.DirectionsService();
      navigator.geolocation.getCurrentPosition(function(position) {
        const origin = {
          lat: position.coords.latitude,
          lng: position.coords.longitude
        };

        const request = {
          origin: origin,
          destination: destination,
          travelMode: google.maps.TravelMode.DRIVING,
          drivingOptions: {
            departureTime: arrivalDateTime,
            trafficModel: 'best_guess'
          }
        };

        service.route(request, function(result, status) {
          if (status === 'OK') {
            const duration = result.routes[0].legs[0].duration.value * 1000;
            const leaveTime = new Date(arrivalDateTime.getTime() - duration);
            const leaveHours = leaveTime.getHours().toString().padStart(2, '0');
            const leaveMinutes = leaveTime.getMinutes().toString().padStart(2, '0');
            resultDiv.textContent = `You should leave by ${leaveHours}:${leaveMinutes} (${leaveTime.toLocaleString()})`;
          } else {
            console.error('Directions API Error:', status);
            resultDiv.textContent = `Could not calculate travel time. Error: ${status}`;
          }
        });
      }, function(error) {
        console.error('Geolocation Error:', error);
        resultDiv.textContent = 'Could not get current location. Make sure location access is allowed.';
      });
    }

    window.onload = initAutocomplete;
  </script>
